39      public void luoTaso() {
40          tiilet = new Tiili[25][25];
41  
42          int x = 0;
43          int y = 0;
44  
45          for (int i = 0; i < KENTTA.length; i++) {
46              if (x == 25) {
47                  x = 0;
48                  y++;
49              }
50              tiilet[x][y] = new Tiili(x * 20, y * 20, KENTTA[i]);
51              x++;
52          }
53  
54      }
