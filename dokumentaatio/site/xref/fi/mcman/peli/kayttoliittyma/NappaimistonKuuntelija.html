16  public class NappaimistonKuuntelija implements KeyListener {
17  
18      private Kayttoliittyma kl;
19      private Pelaaja pelaaja;
20  
21      public NappaimistonKuuntelija(Kayttoliittyma kl) {
22          this.kl = kl;
23          this.pelaaja = kl.getPeli().getPelaaja();
24      }
25  
26      @Override
27      public void keyTyped(KeyEvent e) {
28  
29      }
30  
31      @Override
32      public void keyPressed(KeyEvent e) {
33          if (e.getKeyCode() == KeyEvent.VK_LEFT) {
34              if (pelaaja.voiLiikkuaVasemmalle(pelaaja.getX(), pelaaja.getY())) {
35                  pelaaja.setSuunta(Suunta.VASEN);
36              }
37              pelaaja.liiku();
38          }
39          if (e.getKeyCode() == KeyEvent.VK_RIGHT) {
40              if (pelaaja.voiLiikkuaOikealle(pelaaja.getX(), pelaaja.getY())) {
41                  pelaaja.setSuunta(Suunta.OIKEA);
42              }
43              pelaaja.liiku();
44          }
45          if (e.getKeyCode() == KeyEvent.VK_UP) {
46              if (pelaaja.voiLiikkuaYlos(pelaaja.getX(), pelaaja.getY())) {
47                  pelaaja.setSuunta(Suunta.YLOS);
48              }
49              pelaaja.liiku();
50          }
51          if (e.getKeyCode() == KeyEvent.VK_DOWN) {
52              if (pelaaja.voiLiikkuaAlas(pelaaja.getX(), pelaaja.getY())) {
53                  pelaaja.setSuunta(Suunta.ALAS);
54              }
55  //            } else if  (pelaaja.voiLiikkuaAlas(pelaaja.getX() + 1, pelaaja.getY())) {
56  //                pelaaja.setVaihdaAlas(1);
57  //            } else if  (pelaaja.voiLiikkuaAlas(pelaaja.getX() - 1, pelaaja.getY())) {
58  //                pelaaja.setVaihdaAlas(1);
59  //            } else if  (pelaaja.voiLiikkuaAlas(pelaaja.getX() + 2, pelaaja.getY())) {
60  //                pelaaja.setVaihdaAlas(1);
61  //            } else if  (pelaaja.voiLiikkuaAlas(pelaaja.getX() - 2, pelaaja.getY())) {
62  //                pelaaja.setVaihdaAlas(1);
63  //            }
64              pelaaja.liiku();
65          }
66      }
67  
68      @Override
69      public void keyReleased(KeyEvent e) {
70  
71      }
72  
73  }
