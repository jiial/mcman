118     public void piirraNalkapalkki(Graphics g) {
119         g.setColor(Color.WHITE);
120         int alkuX = 30;
121         int alkuY = 530;
122         g.drawRect(alkuX - 2, alkuY - 2, 388, 19);
123         if (peli.getNalka() < 500) {
124             g.setColor(Color.GREEN);
125         } else if (peli.getNalka() < 1000) {
126             g.setColor(Color.orange);
127         } else {
128             g.setColor(Color.RED);
129         }
130         int neliot = peli.getNalka() / 20;
131         if (peli.getNalka() < 5) {
132             g.fillRect(alkuX, alkuY, 1, 15);
133         } else {
134             for (int i = 0; i < neliot; i++) {
135                 g.fillRect(alkuX, alkuY, 15, 15);
136                 alkuX += 5;
137             }
138         }
139         g.setColor(Color.WHITE);
140         Font fontti = new Font("Arial", Font.PLAIN, 20);
141         g.setFont(fontti);
142         g.drawString("Nälkä", 30, 525);
143     }
